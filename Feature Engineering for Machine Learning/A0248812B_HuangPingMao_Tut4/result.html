def reconstruct_data(coefficient_matrix, eiganvectors):
    points = list()
    coef = np.array(coefficient_matrix)
    n_observations = np.shape(coefficient_matrix)[0]
    n_eigans = np.shape(coefficient_matrix)[1]
    for observation in range(n_observations):
        point = np.zeros(8)
        for eigan_id in range(n_eigans):
            point += coef[observation][eigan_id] * eiganvectors[eigan_id]
        points.append(point)
    return points

def calculate_squared_reconstruction_errors(data1, data2):
    ## using basic operation
    # reconstruction_errors = np.sum(np.square(points - dataset), axis=1)
    ## using dot
    # reconstruction_errors = [np.dot((points - np.array(dataset))[i], (points - np.array(dataset))[i]) for i in range(np.shape(data1)[0])]
    ## using matrix multiplication
    reconstruction_errors = [np.matmul((points - np.array(dataset))[i].T, (points - np.array(dataset))[i]) for i in range(np.shape(data1)[0])]
    return np.array(reconstruction_errors)

def total_squared_reconstruction_errors(data1, data2):
    reconstruction_errors = calculate_squared_reconstruction_errors(data1, data2)
    return np.sum(reconstruction_errors)
