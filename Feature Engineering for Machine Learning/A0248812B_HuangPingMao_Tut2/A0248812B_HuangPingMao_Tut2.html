(ii) Which country gets the longest winning streak in baseball? That is, Which country wins the longest in a row with the top three in baseball?¶

In [9]:
winning_baseball = athlete_events[athlete_events['Sport'] == 'Baseball'].dropna(subset=['Medal'])
winning_baseball = winning_baseball.groupby(['Year', 'Team'])['Medal'].count().unstack().fillna(0)
winning_baseball[winning_baseball > 0] = 2

for row_id in range(len(winning_baseball.index) - 1):
    winning_baseball.iloc[row_id + 1, :] *= winning_baseball.iloc[row_id, :]
winning_baseball[winning_baseball == 0] = 1
winning_baseball = winning_baseball.transform(lambda x: np.log2(x))
winning_baseball.plot(kind='bar')
plt.title('The nuber of consecutive wins in baseball.')

def n_win_team(df, games):
    if df.shape[0] == 0:
        print('No country win', games, 'at the same time.')
        return 0
    else:
        win_team = df.index.to_frame(index=False, name=['Year', 'Team'])
        n_team = win_team.shape[0]
        if n_team == 1:
            print('There is', n_team, 'country wins', games, 'at the same time.')
        elif n_team > 1:
            print('There are', n_team, 'countries win', games, 'at the same time.')
        print(win_team)
    print()
    return n_team
